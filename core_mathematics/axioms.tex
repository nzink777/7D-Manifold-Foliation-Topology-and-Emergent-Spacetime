% =========================================================================
% AXIOM 1
% =========================================================================
\begin{axiom}[The Bulk Manifold]
% [NODE: BULK_SPACE]

\begin{equation}
    \mathcal{M}^7 \cong T^7 = \prod_{A=1}^{7} S^1 = S^1 \times S^1 \times S^1 \times S^1 \times S^1 \times S^1 \times S^1
\end{equation}

\end{axiom}

\begin{axiom}[Structural Objects]
% [NODE: STRUCTURAL_OBJECTS]

\begin{equation}

\end{axiom}
